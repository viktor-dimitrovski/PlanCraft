server {
  listen 443 ssl http2;
  server_name solvebox.org;

  ssl_certificate     /etc/ssl/cloudflare/solvebox.org.crt;
  ssl_certificate_key /etc/ssl/cloudflare/solvebox.org.key;

  access_log /var/log/nginx/solvebox_access.log;
  error_log  /var/log/nginx/solvebox_error.log;

  # --- Landing page ---
  root /var/www/landing;
  index index.html;

  # --- React app under /hive-planner/ ---
  location /hive-planner/ {
    alias /srv/apps/hive-planner/hive-planner-ui/current/;
    index index.html;
    try_files $uri $uri/ /index.html;

    # optional basic auth
    auth_basic "Hive Planner Login";
    auth_basic_user_file /etc/nginx/.htpasswd;
  }

  # --- API under /api/ -> Kestrel (.NET) ---
  location /api/ {
      #allow 203.0.113.25;   # your static IP
      #allow 203.0.113.25;
      #deny all;

      proxy_pass http://127.0.0.1:5000;
      proxy_http_version 1.1;

      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
      proxy_set_header X-Forwarded-Proto $scheme;

      # Basic Auth protection
      #auth_basic "Hive Planner API";
      #auth_basic_user_file /etc/nginx/.htpasswd;

     # CORS headers
      add_header Access-Control-Allow-Origin "https://solvebox.org" always;
      add_header Access-Control-Allow-Credentials true always;
      add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS" always;
      add_header Access-Control-Allow-Headers "Authorization, Content-Type" always;

      # Handle preflight requests directly
      if ($request_method = OPTIONS) {
          add_header Access-Control-Allow-Origin "https://solvebox.org";
          add_header Access-Control-Allow-Credentials true;
          add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS";
          add_header Access-Control-Allow-Headers "Authorization, Content-Type";
          return 204;
      }
  }
  location /api/swagger {
    # optional basic auth
    auth_basic "Hive Planner Login";
    auth_basic_user_file /etc/nginx/.htpasswd;

    proxy_pass http://127.0.0.1:5000/swagger;
  }


}

server {
  listen 80;
  server_name solvebox.org;
  return 301 https://$host$request_uri;
}
